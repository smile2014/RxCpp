216 template<class T>
217 struct average {
218  struct seed_type
219  {
220  seed_type()
221  : value()
222  , count(0)
223  {
224  }
225  T value;
226  int count;
227  rxu::detail::maybe<double> stage;
228  };
229  seed_type seed() {
230  return seed_type{};
231  }
232  seed_type operator()(seed_type& a, T v) {
233  if (a.count != 0 &&
234  (a.count == std::numeric_limits<int>::max() ||
235  ((v > 0) && (a.value > (std::numeric_limits<T>::max() - v))) ||
236  ((v < 0) && (a.value < (std::numeric_limits<T>::min() - v))))) {
237  // would overflow, calc existing and reset for next batch
238  // this will add error to the final result, but the alternative
239  // is to fail on overflow
240  double avg = a.value / a.count;
241  a.value = v;
242  a.count = 1;
243  if (!a.stage.empty()) {
244  a.stage.reset((*a.stage + avg) / 2);
245  } else {
246  a.stage.reset(avg);
247  }
248  } else {
249  a.value += v;
250  ++a.count;
251  }
252  return a;
253  }
254  double operator()(seed_type& a) {
255  if (a.count > 0) {
256  double avg = a.value / a.count;
257  if (!a.stage.empty()) {
258  avg = (*a.stage + avg) / 2;
259  }
260  return avg;
261  }
262  throw rxcpp::empty_error("average() requires a stream with at least one value");
263  }
264 };
